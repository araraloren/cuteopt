Ctx hold all the OptKeepers, provide the inteface parse the command line arguments.

use cuteopt::prelude::*;
 
#[derive(Debug, Clone, Eq, PartialEq)]
enum ParseState {
    PSBoolean,
    PSString,
    PSDefault,
}
 
impl Default for ParseState {
    fn default() -> Self {
        Self::PSDefault
    }
}
 
let mut ctx = Ctx::new();
 
ctx.add_bool("--boolean", ParseState::PSBoolean);
ctx.add_str("--string", ParseState::PSString);
 
ctx.parse(&mut std::env::args().skip(1));
